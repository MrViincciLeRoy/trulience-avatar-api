  <script>
    let trulience = null;
    let micEnabled = false;
    let cfg = {};

    function setStatus(text, live) {
      document.getElementById('status-text').textContent = text;
      const dot = document.getElementById('status-dot');
      dot.className = 'status-dot' + (live ? ' live' : '');
    }

    function setConnectedUI(connected) {
      document.getElementById('btn-connect').disabled = connected;
      document.getElementById('btn-disconnect').disabled = !connected;
      document.getElementById('btn-mic').disabled = !connected;
      document.getElementById('textBox').disabled = !connected;
      document.getElementById('btn-send').disabled = !connected;
      if (connected) {
        document.getElementById('video-placeholder').style.display = 'none';
      }
    }

    function initAvatar() {
      const avatarId = document.getElementById('inp-avatar-id').value.trim();
      const token    = document.getElementById('inp-token').value.trim();
      if (!avatarId) { alert('Please enter an Avatar ID.'); return; }
      cfg = { avatarId, token };
      document.getElementById('config-screen').style.display = 'none';
      document.getElementById('avatar-screen').style.display = 'flex';
      setStatus('Ready', false);
    }

    function startCall() {
      setStatus('Connecting…', false);

      const builder = new TrulienceBuilder()
        .setAvatarId(cfg.avatarId)
        .setVideoElement(document.getElementById('myvideo'));

      if (cfg.token) builder.setToken(cfg.token);

      builder.setEventCallback('auth-success', () => {
        setStatus('Live', true);
        setConnectedUI(true);
        micEnabled = false;
        updateMicBtn();
      });

      builder.setEventCallback('auth-fail', (err) => {
        setStatus('Auth failed', false);
        console.error(err);
      });

      builder.setEventCallback('trl-avatar-disconnected', () => {
        setStatus('Disconnected', false);
        setConnectedUI(false);
        document.getElementById('video-placeholder').style.display = 'flex';
      });

      trulience = builder.build();
      trulience.connect();
    }

    function endCall() {
      if (trulience) { trulience.disconnect(); trulience = null; }
      setStatus('Disconnected', false);
      setConnectedUI(false);
      document.getElementById('video-placeholder').style.display = 'flex';
    }

    function updateMicBtn() {
      const btn = document.getElementById('btn-mic');
      if (micEnabled) {
        btn.textContent = 'Mute';
        btn.classList.add('muted-active');
      } else {
        btn.textContent = 'Mic off';
        btn.classList.remove('muted-active');
      }
    }

    function toggleMic() {
      if (!trulience) return;
      micEnabled = !micEnabled;
      trulience.setMicEnabled(micEnabled);
      updateMicBtn();
    }

    function sendText() {
      const box = document.getElementById('textBox');
      const msg = box.value.trim();
      if (!msg || !trulience) return;
      trulience.sendMessage(msg);
      box.value = '';
    }
  </script>
